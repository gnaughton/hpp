var sMax;	// Isthe maximum number of stars
var preSet; // Is the PreSet value onces a selection has been made
var rated;

//Switch on the stars depending on which one we're hovering over.

function hoverOver(num){

    //Get the total number of stars.
	sMax = 0;	
	for(n=0; n<num.parentNode.childNodes.length; n++){
		if(num.parentNode.childNodes[n].nodeName == "A"){
			sMax++;	
		}
	}
	
	//Loop through and switch on stars.
	s = num.id.replace("_", ''); 
	for(i=1; i<=sMax; i++){		
		if(i<=s){
			document.getElementById("_"+i).className = "on";
		}else{
			document.getElementById("_"+i).className = "";
			}
		}
	
    //Set the rating value in the interface.	
	document.getElementById("rateStatus").innerHTML = num.title;	
	
}

// For when you roll out of the the whole thing //
function hoverOff(me){
	if(!preSet){	
	    
		//No rating, so switch off all the stars.
		for(i=1; i<=sMax; i++){		
			document.getElementById("_"+i).className = "";
			document.getElementById("rateStatus").innerHTML = me.parentNode.title;
		}
	
	}
	else{
	
	    //Switch on the stars up to the rating.
		hoverOver(preSet);
		//document.getElementById("rateStatus").innerHTML = document.getElementById("ratingSaved").innerHTML;
	}
}


// When you actually rate something //
function setRating(me){
	
	//Set this link as the current rating.
	preSet = me;
	
	//Set the rating in the input field.
	document.getElementById("entry_10").value = preSet.id.replace("_", '');
    
	
	//Switch on the stars up to the current rating.
	hoverOver(preSet);
	
}
